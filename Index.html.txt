  <script>
    // --- Helpers ---
    const $$ = (sel, root=document) => Array.from(root.querySelectorAll(sel));
    const storageKey = "wishlist_bought_v1";

    // --- Bought state (localStorage) ---
    const saved = JSON.parse(localStorage.getItem(storageKey) || "{}");

    $$(".card li").forEach((li, idx) => {
      const id = li.dataset.name || ("item_" + idx);
      li.dataset.id = id;
      const chk = li.querySelector(".chk");
      chk.checked = !!saved[id];
      if (chk.checked) li.classList.add("bought");

      chk.addEventListener("change", () => {
        saved[id] = chk.checked;
        localStorage.setItem(storageKey, JSON.stringify(saved));
        li.classList.toggle("bought", chk.checked);
      });
    });

    // --- Search filter ---
    const search = document.getElementById("search");
    search.addEventListener("input", () => {
      const q = search.value.trim().toLowerCase();
      $$(".card li").forEach(li => {
        const hay = (li.dataset.name + " " + (li.dataset.tags || "")).toLowerCase();
        li.style.display = hay.includes(q) ? "" : "none";
      });
      // hide empty sections
      $$(".card").forEach(card => {
        const anyVisible = $$("li", card).some(li => li.style.display !== "none");
        card.style.display = anyVisible ? "" : "none";
      });
    });

    // --- Copy list ---
    document.getElementById("copyBtn").addEventListener("click", async () => {
      const sections = $$(".card").map(card => {
        const title = card.getAttribute("data-section") || card.querySelector("h2")?.innerText?.trim() || "Section";
        const items = $$("li", card)
          .map(li => li.querySelector(".name")?.innerText?.trim())
          .filter(Boolean);
        return { title, items };
      }).filter(s => s.items.length);

      const text = sections.map(s => {
        return `‚Ä¢ ${s.title}\n` + s.items.map(i => `  - ${i}`).join("\n");
      }).join("\n\n");

      try{
        await navigator.clipboard.writeText(text);
        alert("Copied!");
      }catch(e){
        prompt("Copy this text:", text);
      }
    });

    // --- Reset bought ---
    document.getElementById("resetBtn").addEventListener("click", () => {
      if (!confirm("Clear all bought checkmarks on this device?")) return;
      localStorage.removeItem(storageKey);
      $$(".card li").forEach(li => {
        const chk = li.querySelector(".chk");
        chk.checked = false;
        li.classList.remove("bought");
      });
    });
  </script>
